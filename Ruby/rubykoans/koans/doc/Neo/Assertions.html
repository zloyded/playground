def assert_raise(exception)
  begin
    yield
  rescue Exception => ex
    expected = ex.is_a?(exception)
    assert(expected, "Exception #{exception.inspect} expected, but #{ex.inspect} was raised")
    return ex
  end
  flunk "Exception #{exception.inspect} expected, but nothing raised"
end
