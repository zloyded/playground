def test_if_statements_return_values
  value = if true
            :true_value
          else
            :false_value
          end
  assert_equal __, value

  value = if false
            :true_value
          else
            :false_value
          end
  assert_equal __, value

  # NOTE: Actually, EVERY statement in Ruby will return a value, not
  # just if statements.
end
